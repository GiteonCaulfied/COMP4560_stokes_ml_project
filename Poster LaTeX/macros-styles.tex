% this macro creates a content block for the poster
% 
% usage: \PBlock{id}{node-params}{heading}{content}
% 
\newcommand{\PBlock}[4]{
\begin{tikzpicture}
\node[box] (#1) {\begin{minipage}{\columnwidth-2em}#4\end{minipage}};
\node[heading] at (#1.north west) {#3};
\end{tikzpicture}
}

\newcommand{\Block}[2]{
  \PBlock{some-label}{}{#1}{#2}
}
